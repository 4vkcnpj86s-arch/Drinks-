    <script>
        const STORAGE_KEY = 'DRINK_JOURNAL_2026_PURPLE_V1';
        const months = ["JAN ä¸€æœˆ", "FEB äºŒæœˆ", "MAR ä¸‰æœˆ", "APR å››æœˆ", "MAY äº”æœˆ", "JUN å…­æœˆ", "JUL ä¸ƒæœˆ", "AUG å…«æœˆ", "SEP ä¹æœˆ", "OCT åæœˆ", "NOV åä¸€æœˆ", "DEC åäºŒæœˆ"];
        const weekdays = ["SUN", "MON", "TUE", "WED", "THU", "FRI", "SAT"];
        const sugarOptions = ["ç„¡æ³•", "å…¨ç³–", "åŠç³–", "å¾®ç³–", "1åˆ†", "ç„¡ç³–"];
        const iceOptions = ["å°‘å†°", "å¾®å†°", "å»å†°", "å¸¸æº«", "æº«"];
        
        // ä½é£½å’Œåº¦ç´«è‰²ç³»è‰²ç›¤
        const purplePalette = ['#9b8ea9', '#a79ab4', '#b9b2c3', '#8675a9', '#7a6a99', '#c5bbd1', '#dcd6e0', '#6b5e80', '#b1abb9'];

        let currentMonth = new Date().getMonth(); 
        let activeDay = null;
        let drinkData = JSON.parse(localStorage.getItem(STORAGE_KEY)) || {};
        let monthChartInstance = null;
        let yearChartInstance = null;

        function initData() {
            if (Object.keys(drinkData).length === 0) {
                for (let m = 0; m < 12; m++) drinkData[m] = {};
            }
        }

        function getCupSvg(color) {
            return `
                <svg viewBox="0 0 100 100" fill="none" xmlns="http://www.w3.org/2000/svg">
                    <path d="M25 20L32 85C32.5 89.5 36.5 93 41 93H59C63.5 93 67.5 89.5 68 85L75 20H25Z" fill="${color}" fill-opacity="0.3" stroke="${color}" stroke-width="4"/>
                    <path d="M20 20H80V28H20V20Z" fill="${color}"/>
                    <path d="M60 10L55 20H65L60 10Z" fill="${color}"/>
                    <circle cx="40" cy="50" r="3" fill="${color}" fill-opacity="0.6"/>
                    <circle cx="55" cy="65" r="3" fill="${color}" fill-opacity="0.6"/>
                    <circle cx="48" cy="75" r="3" fill="${color}" fill-opacity="0.6"/>
                </svg>
            `;
        }

        function renderMonthNav() {
            const nav = document.getElementById('month-nav');
            nav.innerHTML = '';
            months.forEach((name, idx) => {
                const btn = document.createElement('button');
                btn.className = `month-btn ${idx === currentMonth ? 'active' : ''}`;
                btn.textContent = name;
                btn.onclick = () => { currentMonth = idx; renderCalendar(); renderMonthNav(); };
                nav.appendChild(btn);
            });
        }

        function renderCalendar() {
            const container = document.getElementById('calendar-container');
            container.innerHTML = '';
            document.getElementById('current-month-title').textContent = months[currentMonth];

            weekdays.forEach(day => {
                const d = document.createElement('div');
                d.className = 'weekday-label';
                d.textContent = day;
                container.appendChild(d);
            });

            const daysCount = new Date(2026, currentMonth + 1, 0).getDate();
            const firstDay = new Date(2026, currentMonth, 1).getDay();

            for (let i = 0; i < firstDay; i++) container.appendChild(document.createElement('div'));

            for (let d = 1; d <= daysCount; d++) {
                const cell = document.createElement('div');
                cell.className = `day-cell`;
                cell.onclick = () => openModal(d);
                
                const validDrinks = (drinkData[currentMonth][d] || []).filter(dr => dr.shop || dr.item);
                const count = validDrinks.length;
                if (count > 0) cell.classList.add('has-data');

                let sizeClass = count >= 3 ? 'cup-size-3' : (count === 2 ? 'cup-size-2' : 'cup-size-1');
                let iconsHtml = validDrinks.map((_, k) => `<span class="cup-svg-wrapper ${sizeClass}">${getCupSvg(purplePalette[k % purplePalette.length])}</span>`).join('');
                
                cell.innerHTML = `<div class="day-num">${d.toString().padStart(2, '0')}</div><div class="drink-icons-container">${iconsHtml}</div>`;
                container.appendChild(cell);
            }
            calculateStats();
        }

        function openModal(day) {
            activeDay = day;
            document.getElementById('modal-title').textContent = `${months[currentMonth].split(' ')[1]} ${day}æ—¥`;
            document.getElementById('modal-overlay').style.display = 'flex';
            const container = document.getElementById('drink-forms-container');
            container.innerHTML = '';
            if (!drinkData[currentMonth][day]) drinkData[currentMonth][day] = Array(3).fill().map(() => ({shop:'', item:'', price:'', sugar:'', ice:''}));
            
            drinkData[currentMonth][day].forEach((drink, i) => {
                const section = document.createElement('div');
                section.className = 'drink-section';
                section.innerHTML = `
                    <div class="grid grid-cols-2 gap-3 mb-3">
                        <input type="text" class="input-field" placeholder="åº—å" value="${drink.shop}" oninput="updateDrink(${i}, 'shop', this.value)">
                        <input type="text" class="input-field" placeholder="å“é …" value="${drink.item}" oninput="updateDrink(${i}, 'item', this.value)">
                    </div>
                    <input type="number" class="input-field mb-3" placeholder="é‡‘é¡" value="${drink.price}" oninput="updateDrink(${i}, 'price', this.value)">
                    <div class="grid grid-cols-6 gap-1 mb-2">${sugarOptions.map(opt => `<div class="option-btn ${drink.sugar === opt ? 'active' : ''}" onclick="toggleOpt(${i}, 'sugar', '${opt}', this)">${opt}</div>`).join('')}</div>
                    <div class="grid grid-cols-5 gap-1">${iceOptions.map(opt => `<div class="option-btn ${drink.ice === opt ? 'active' : ''}" onclick="toggleOpt(${i}, 'ice', '${opt}', this)">${opt}</div>`).join('')}</div>
                `;
                container.appendChild(section);
            });
        }

        function updateDrink(idx, field, val) { drinkData[currentMonth][activeDay][idx][field] = val; save(); }
        function toggleOpt(idx, field, val, el) {
            const current = drinkData[currentMonth][activeDay][idx][field];
            drinkData[currentMonth][activeDay][idx][field] = (current === val) ? '' : val;
            save();
            const btns = el.parentElement.querySelectorAll('.option-btn');
            btns.forEach(b => b.classList.remove('active'));
            if (drinkData[currentMonth][activeDay][idx][field]) el.classList.add('active');
        }
        function save() { localStorage.setItem(STORAGE_KEY, JSON.stringify(drinkData)); }

        function calculateStats() {
            let totalCups = 0, totalCost = 0, mCups = 0, mCost = 0;
            let mShopStats = {}, yShopStats = {};

            for (let m = 0; m < 12; m++) {
                for (let d in drinkData[m]) {
                    drinkData[m][d].forEach(dr => {
                        if (dr.shop || dr.item) {
                            const p = parseInt(dr.price) || 0;
                            totalCups++; totalCost += p;
                            const shop = dr.shop || 'æœªåˆ†é¡';
                            yShopStats[shop] = (yShopStats[shop] || 0) + p;
                            if (parseInt(m) === currentMonth) { 
                                mCups++; mCost += p; 
                                mShopStats[shop] = (mShopStats[shop] || 0) + p;
                            }
                        }
                    });
                }
            }
            document.getElementById('month-total-cups').textContent = mCups;
            document.getElementById('month-total-cost').textContent = mCost;
            document.getElementById('year-total-info').innerHTML = `å¹´åº¦é£²ç”¨: <span class="stat-value">${totalCups}</span> æ¯&nbsp;&nbsp;&nbsp;ç¸½æ¶ˆè²»: <span class="stat-value">${totalCost.toLocaleString()}</span> å…ƒ`;
            
            monthChartInstance = renderPieChart('monthChart', mShopStats, monthChartInstance, 'no-month-data');
            yearChartInstance = renderPieChart('yearChart', yShopStats, yearChartInstance, 'no-year-data');
        }

        function renderPieChart(canvasId, stats, chartInstance, noDataId) {
            const labels = Object.keys(stats);
            const data = Object.values(stats);
            const canvas = document.getElementById(canvasId);
            const noData = document.getElementById(noDataId);

            if (labels.length === 0) {
                noData.classList.remove('hidden');
                canvas.classList.add('hidden');
                return chartInstance;
            } else {
                noData.classList.add('hidden');
                canvas.classList.remove('hidden');
            }

            if (chartInstance) {
                chartInstance.data.labels = labels;
                chartInstance.data.datasets[0].data = data;
                chartInstance.update();
                return chartInstance;
            } else {
                return new Chart(canvas.getContext('2d'), {
                    type: 'pie',
                    data: {
                        labels: labels,
                        datasets: [{ 
                            data: data, 
                            backgroundColor: purplePalette, 
                            borderWidth: 2, 
                            borderColor: '#ffffff' 
                        }]
                    },
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        plugins: {
                            legend: { display: false },
                            tooltip: {
                                bodyFont: { family: 'ChenYuluoyan', size: 14 },
                                callbacks: {
                                    label: (ctx) => ` ${ctx.label}: $${ctx.parsed} (${Math.round(ctx.parsed / ctx.dataset.data.reduce((a,b)=>a+b,0) * 100)}%)`
                                }
                            }
                        }
                    }
                });
            }
        }

        function closeModal(e) {
            if (e && e.target !== document.getElementById('modal-overlay')) return;
            document.getElementById('modal-overlay').style.display = 'none';
            renderCalendar();
        }

        initData();
        renderMonthNav();
        renderCalendar();
    </script>
